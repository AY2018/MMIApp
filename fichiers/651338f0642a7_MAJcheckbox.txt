<?php
session_start();

if (isset($_POST['devoirID']) && isset($_POST['isChecked'])) {
    $devoirID = $_POST['devoirID'];
    $isChecked = $_POST['isChecked'] ? $_SESSION['id'] : null;

    // Connexion à la base de données (remplacez les informations de connexion)
    $link = mysqli_connect("localhost", "nlerond_utilisateur", "utilisateur123", "nlerond_mmiapp");
    
    if (!$link) {
        die("Erreur de connexion à la base de données : " . mysqli_connect_error());
    }

    // Mettez à jour la colonne isDone dans la table devoirs
    $sql = "UPDATE devoirs SET isDone = $isChecked WHERE idDevoir = $devoirID";
    $result = mysqli_query($link, $sql);

    if (!$result) {
        echo "Erreur lors de la mise à jour du devoir : " . mysqli_error($link);
    }

    // Fermez la connexion à la base de données
    mysqli_close($link);
